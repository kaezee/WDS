<script>
function toggleAccordion(button) {
  const item = button.closest('.accordion-item');
  const wasActive = item.classList.contains('active');
  
  // Single-open mode: close all others
  document.querySelectorAll('.accordion-item').forEach(i => i.classList.remove('active'));
  
  // Toggle current item
  if (!wasActive) {
    item.classList.add('active');
  }
}
</script>
import { useState } from 'react';

const Accordion = ({ items, allowMultiple = false }) => {
  const [openItems, setOpenItems] = useState(new Set([0])); // First item open by default

  const toggleItem = (index) => {
    setOpenItems((prev) => {
      const newSet = new Set(allowMultiple ? prev : []);
      if (prev.has(index)) {
        newSet.delete(index);
      } else {
        newSet.add(index);
      }
      return newSet;
    });
  };

  return (
    <div className="accordion">
      {items.map((item, index) => (
        <div
          key={index}
          className={`accordion-item ${openItems.has(index) ? 'active' : ''}`}
        >
          <button
            className="accordion-header"
            onClick={() => toggleItem(index)}
          >
            <span className="accordion-header-text">{item.title}</span>
            <svg className="accordion-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M19 9l-7 7-7-7" />
            </svg>
          </button>
          <div className="accordion-content">
            <div className="accordion-body">{item.content}</div>
          </div>
        </div>
      ))}
    </div>
  );
};

// Usage
<Accordion
  items={[
    { title: 'What is your return policy?', content: 'We accept returns...' },
    { title: 'How do I track my order?', content: 'Once your order ships...' }
  ]}
  allowMultiple={false}
/>

Responsive Behavior

Accordions automatically adapt to different screen sizes with optimized padding and font sizes.

Breakpoint Adjustments:

  • Mobile (≤640px): Reduced padding (16px → 12px), smaller text (16px → 14px), smaller icons (20px → 18px)
  • Tablet (641-1024px): Medium padding (20px), standard text (16px), standard icons (20px)
  • Desktop (≥1025px): Full padding (24px), standard text (16px), standard icons (20px)

Text automatically wraps on smaller screens, and touch targets remain accessible at minimum 44x44px.
