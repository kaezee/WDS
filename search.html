Search Bar

Help users find products, suppliers, and content quickly.

Usage & Purpose

Use search bars when: Users need to find specific items in large datasets, navigate product catalogs, or locate suppliers and orders.

Don't use search bars for: Filtering within small lists (use dropdown), navigation (use menu), or when all options can be displayed at once.

Basic Search

Standard search input with icon and clear button.

HTML + CSS 
<div class="search" id="search-1">
  <span class="search-icon"></span>
  <input 
    type="text" 
    class="search-input" 
    placeholder="Search products..."
    oninput="toggleClear('search-1')"
  >
  <button class="search-clear" onclick="clearSearch('search-1')">
    <span class="search-clear-icon"></span>
  </button>
</div>

<script>
function toggleClear(id) {
  const search = document.getElementById(id);
  const input = search.querySelector('.search-input');
  if (input.value.length > 0) {
    search.classList.add('has-value');
  } else {
    search.classList.remove('has-value');
  }
}

function clearSearch(id) {
  const search = document.getElementById(id);
  const input = search.querySelector('.search-input');
  input.value = '';
  search.classList.remove('has-value');
  input.focus();
}
</script>
JSX 
const SearchBar = ({ placeholder, onSearch }) => {
  const [value, setValue] = useState('');

  const handleClear = () => {
    setValue('');
  };

  return (
    <div className={`search ${value ? 'has-value' : ''}`}>
      <span className="search-icon"></span>
      <input
        type="text"
        className="search-input"
        placeholder={placeholder}
        value={value}
        onChange={(e) => setValue(e.target.value)}
      />
      <button className="search-clear" onClick={handleClear}>
        <span className="search-clear-icon"></span>
      </button>
    </div>
  );
};

<!-- Add 'open' class to show suggestions -->
JSX 
const SearchWithSuggestions = ({ suggestions }) => {
  const [query, setQuery] = useState('');
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className={`search ${isOpen ? 'open' : ''}`}>
      <span className="search-icon"></span>
      <input
        type="text"
        className="search-input"
        placeholder="Search products..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        onFocus={() => setIsOpen(true)}
        onBlur={() => setTimeout(() => setIsOpen(false), 200)}
      />
      
      <div className="search-suggestions">
        {suggestions.map((group, idx) => (
          <div key={idx} className="search-suggestion-group">
            <div className="search-suggestion-title">{group.title}</div>
            {group.items.map((item, i) => (
              <a key={i} href={item.href} className="search-suggestion-item">
                <span className="search-suggestion-icon"></span>
                <span className="search-suggestion-text">{item.text}</span>
                <span className="search-suggestion-meta">{item.meta}</span>
              </a>
            ))}
          </div>
        ))}
      </div>
    </div>
  );
};
