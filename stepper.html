Stepper

Progress indicators for multi-step processes like checkout flows, onboarding, and quote requests.

Usage & Purpose

Horizontal Stepper

Default layout for desktop checkout flows and onboarding processes.

Vertical Stepper

For detailed forms with content between steps, better for mobile layouts.

Compact Stepper

Space-saving variant for sidebars, modals, or quick 3-4 step processes.

With Icons

Visual indicators replacing numbers for branded flows and order tracking.

With Descriptions

Add helper text for complex processes needing explanation at each step.

const HorizontalStepper = ({ steps, currentStep }) => (
  <div className="stepper">
    {steps.map((step, index) => {
      const stepNumber = index + 1;
      const isCompleted = stepNumber < currentStep;
      const isActive = stepNumber === currentStep;
      
      return (
        <React.Fragment key={index}>
          <div className={`stepper-step ${isCompleted ? 'completed' : ''} ${isActive ? 'active' : ''}`}>
            <div className="stepper-circle">
              {isCompleted ? '✓' : stepNumber}
            </div>
            <span className="stepper-label">{step.label}</span>
          </div>
          
          {index < steps.length - 1 && (
            <div className={`stepper-line ${isCompleted ? 'completed' : ''}`} />
          )}
        </React.Fragment>
      );
    })}
  </div>
);

// Usage
const checkoutSteps = [
  { label: 'Product Details' },
  { label: 'Shipping Info' },
  { label: 'Payment' },
  { label: 'Confirmation' }
];

<HorizontalStepper steps={checkoutSteps} currentStep={2} />

const VerticalStepper = ({ steps, currentStep }) => (
  <div className="stepper stepper-vertical">
    {steps.map((step, index) => {
      const stepNumber = index + 1;
      const isCompleted = stepNumber < currentStep;
      const isActive = stepNumber === currentStep;
      const isLast = index === steps.length - 1;
      
      return (
        <div key={index} className={`stepper-step ${isCompleted ? 'completed' : ''} ${isActive ? 'active' : ''}`}>
          <div className="stepper-circle">
            {isCompleted ? '✓' : stepNumber}
          </div>
          <div>
            <span className="stepper-label">{step.label}</span>
          </div>
          {!isLast && <div className={`stepper-line ${isCompleted ? 'completed' : ''}`} />}
        </div>
      );
    })}
  </div>
);

import { Check, Package, Truck, Home } from 'lucide-react';

const IconStepper = ({ steps, currentStep }) => (
  <div className="stepper stepper-icon">
    {steps.map((step, index) => {
      const Icon = step.icon;
      const stepNumber = index + 1;
      const isCompleted = stepNumber < currentStep;
      const isActive = stepNumber === currentStep;
      
      return (
        <React.Fragment key={index}>
          <div className={`stepper-step ${isCompleted ? 'completed' : ''} ${isActive ? 'active' : ''}`}>
            <div className="stepper-circle">
              <Icon />
            </div>
            <span className="stepper-label">{step.label}</span>
          </div>
          {index < steps.length - 1 && (
            <div className={`stepper-line ${isCompleted ? 'completed' : ''}`} />
          )}
        </React.Fragment>
      );
    })}
  </div>
);

// Usage
const orderSteps = [
  { label: 'Order Placed', icon: Check },
  { label: 'Processing', icon: Package },
  { label: 'In Transit', icon: Truck },
  { label: 'Delivered', icon: Home }
];

<IconStepper steps={orderSteps} currentStep={3} />

const DescriptionStepper = ({ steps, currentStep }) => (
  <div className="stepper stepper-vertical">
    {steps.map((step, index) => {
      const stepNumber = index + 1;
      const isCompleted = stepNumber < currentStep;
      const isActive = stepNumber === currentStep;
      const isLast = index === steps.length - 1;
      
      return (
        <div key={index} className={`stepper-step ${isCompleted ? 'completed' : ''} ${isActive ? 'active' : ''}`}>
          <div className="stepper-circle">
            {isCompleted ? '✓' : stepNumber}
          </div>
          <div>
            <span className="stepper-label">{step.label}</span>
            <span className="stepper-description">{step.description}</span>
          </div>
          {!isLast && <div className={`stepper-line ${isCompleted ? 'completed' : ''}`} />}
        </div>
      );
    })}
  </div>
);
