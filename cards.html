Cards

Container components for marketplace listings, content, and information display.

Usage & Purpose

Use cards when: Displaying products in marketplace grids, showing supplier profiles, organizing content like blog posts, or presenting structured information in dashboards.

Don't use cards for: Single pieces of information that don't need visual separation, or when a simple list would be clearer.

Product Cards

Main marketplace listing cards with 4:3 images, no public pricing, and request quote CTAs. Display product details, supplier info, certifications, and attributes.

JSX 
const ProductCard = ({ 
  product, 
  label, 
  certifications = [], 
  attributes = [] 
}) => {
  return (
    <div className="card-product">
      <div className="card-product-image">
        {label && (
          <span className={`card-product-label card-product-label-${label.type}`}>
            {label.text}
          </span>
        )}
        <img src={product.image} alt={product.name} />
      </div>
      <div className="card-product-content">
        <h3 className="card-product-title">{product.name}</h3>
        <p className="card-product-supplier">{product.supplier}</p>
        
        {certifications.length > 0 && (
          <div className="card-product-certifications">
            {certifications.map(cert => (
              <span key={cert} className={`badge badge-cert-${cert.toLowerCase()}`}>
                {cert}
              </span>
            ))}
          </div>
        )}
        
        {attributes.length > 0 && (
          <div className="card-product-attributes">
            {attributes.map(attr => (
              <span 
                key={attr.text}
                className={`card-product-attribute card-product-attribute-${attr.type}`}
              >
                <span className="card-product-attribute-icon"></span>
                {attr.text}
              </span>
            ))}
          </div>
        )}
        
        <p className="card-product-min-order">Min order: {product.minOrder} kg</p>
        
        <button className="btn btn-primary btn-md card-product-cta">
          Request Quote
        </button>
      </div>
    </div>
  );
};

// Usage
<ProductCard 
  product={{
    name: "Atlantic Salmon Fillets",
    supplier: "Norwegian Seafoods Co.",
    image: "/images/salmon.jpg",
    minOrder: 50
  }}
  label={{ type: 'new', text: 'New' }}
  certifications={['MSC', 'ASC']}
  attributes={[
    { type: 'popular', text: 'Popular' }
  ]}
/>

const CategoryCard = ({ category }) => {
  return (
    <div className="card-category">
      <div className="card-category-icon">
        {category.icon ? (
          <img src={category.icon} alt={category.name} />
        ) : (
          <span className="card-category-icon-placeholder"></span>
        )}
      </div>
      <h3 className="card-category-title">{category.name}</h3>
      <p className="card-category-count">{category.count} products</p>
    </div>
  );
};
