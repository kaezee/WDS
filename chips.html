Chips & Tags

Compact elements for filters, labels, selections, and counts in marketplace interfaces.

Usage & Purpose

Use chips when: Displaying active filters that users can remove, enabling multi-select category browsing, or indicating numerical counts.

Don't use chips for: Product attributes or certifications (use Badge component instead), primary actions (use buttons), or long text content.

const FilterChip = ({ label, onRemove, hasIcon, size = 'default' }) => {
  const sizeClass = size !== 'default' ? `chip-filter-${size}` : '';
  
  return (
    <span className={`chip-filter ${sizeClass}`.trim()}>
      {hasIcon && <span className="chip-filter-icon"></span>}
      {label}
      <button className="chip-filter-remove" onClick={onRemove}>✕</button>
    </span>
  );
};

JSX 
const InteractiveChip = ({ label, selected, onToggle, hasIcon }) => {
  return (
    <span 
      className={`chip-interactive ${selected ? 'selected' : ''}`.trim()}
      onClick={onToggle}
    >
      {hasIcon && <span className="chip-interactive-icon"></span>}
      {label}
    </span>
  );
};

// Usage with State
const CategoryFilter = () => {
  const [selected, setSelected] = useState(['Salmon', 'Cod']);
  
  const categories = ['Salmon', 'Tuna', 'Cod', 'Shellfish', 'Lobster', 'Crab'];
  
  const toggleCategory = (category) => {
    setSelected(prev => 
      prev.includes(category) 
        ? prev.filter(c => c !== category)
        : [...prev, category]
    );
  };
  
  return (
    <div className="chip-group">
      {categories.map(cat => (
        <InteractiveChip
          key={cat}
          label={cat}
          selected={selected.includes(cat)}
          onToggle={() => toggleCategory(cat)}
        />
      ))}
    </div>
  );
};

JSX 
const CountBadge = ({ count, variant = 'default', size = 'default' }) => {
  const variantClass = variant !== 'default' ? `chip-count-${variant}` : '';
  const sizeClass = size === 'lg' ? 'chip-count-lg' : '';
  
  // Format large numbers
  const displayCount = count > 99 ? '99+' : count;
  
  return (
    <span className={`chip-count ${variantClass} ${sizeClass}`.trim()}>
      {displayCount}
    </span>
  );
};

// Usage
