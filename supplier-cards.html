Supplier Cards

Display supplier profiles with logos, locations, certifications, and contact actions.

Usage & Purpose

Use supplier cards when: Displaying supplier directories, search results, recommended suppliers, or in supplier comparison views.

Don't use supplier cards for: Individual product listings (use Product Cards instead) or detailed supplier profiles (use full page layout).

Supplier Card Variants

Cards showing supplier information with logo, company name, location, and certifications.

JSX 
const SupplierCard = ({ supplier }) => {
  return (
    <div className="card-supplier">
      <div className="card-supplier-header">
        <div className="card-supplier-logo">
          {supplier.logo ? (
            <img src={supplier.logo} alt={supplier.name} />
          ) : (
            <span className="card-supplier-logo-placeholder"></span>
          )}
        </div>
      </div>
      <div className="card-supplier-content">
        <h3 className="card-supplier-name">{supplier.name}</h3>
        <div className="card-supplier-location">
          <span className="card-supplier-location-icon"></span>
          {supplier.location}
        </div>
        {supplier.certifications && supplier.certifications.length > 0 && (
          <div className="card-supplier-certifications">
            {supplier.certifications.map(cert => (
              <span key={cert.name} className={`cert-badge cert-badge-${cert.color}`}>
                <span className="cert-badge-icon"></span>
                {cert.name}
              </span>
            ))}
          </div>
        )}
        <button className="btn btn-md btn-primary-outline card-supplier-cta">
          View Profile
        </button>
      </div>
    </div>
  );
};

// Usage
<SupplierCard 
  supplier={{
    name: "Norwegian Seafoods Co.",
    location: "Bergen, Norway",
    logo: "/logos/norwegian-seafoods.png",
    certifications: [
      { name: 'MSC', color: 'blue' },
      { name: 'ASC', color: 'blue' },
      { name: 'HACCP', color: 'gray' }
    ]
  }}
/>

// Grid Layout
<div className="card-grid">
  {suppliers.map(supplier => (
    <SupplierCard key={supplier.id} supplier={supplier} />
  ))}
</div>
