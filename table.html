Table

Display structured data in rows and columns for inventory, orders, and listings.

Usage & Purpose

Use tables when: Displaying structured data that needs to be scanned, compared, or sorted, like order history, inventory lists, or supplier comparisons.

Don't use tables for: Layout purposes (use CSS Grid), small datasets better suited for cards, or content that doesn't need row/column structure.

JSX 
const Table = ({ data }) => {
  return (
    <div className="table-container">
      <table className="table">
        <thead>
          <tr>
            <th>Product</th>
            <th>Supplier</th>
            <th className="numeric">Quantity</th>
            <th className="numeric">Price</th>
            <th>Status</th>
          </tr>
        </thead>
        <tbody>
          {data.map((row, index) => (
            <tr key={index}>
              <td className="primary">{row.product}</td>
              <td>{row.supplier}</td>
              <td className="numeric">{row.quantity} kg</td>
              <td className="numeric">${row.price.toLocaleString()}</td>
              <td className="status">
                <span className={`order-status order-status-${row.status.toLowerCase()}`}>
                  {row.status}
                </span>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
};

JSX 
const [sortConfig, setSortConfig] = useState({ key: null, direction: null });

const handleSort = (key) => {
  let direction = 'asc';
  if (sortConfig.key === key && sortConfig.direction === 'asc') {
    direction = 'desc';
  }
  setSortConfig({ key, direction });
};

const getSortClass = (key) => {
  if (sortConfig.key !== key) return 'sortable';
  return `sortable sort-${sortConfig.direction}`;
};

<th 
  className={getSortClass('amount')} 
  onClick={() => handleSort('amount')}
>
  Amount
</th>
