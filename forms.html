Input Fields & Forms

Text inputs, textareas, select dropdowns, checkboxes, and radio buttons for capturing user data in B2B workflows.

Usage & Purpose

Text Input

Single-line fields for names, emails, search queries, and short-form data entry.

Textarea

Multi-line fields for descriptions, notes, comments, and long-form content.

Select Dropdown

Single-choice selection from predefined options like categories, regions, or statuses.

Checkbox

Binary choices or multiple selections like terms acceptance, feature toggles, or filters.

Radio Button

Exclusive single selection from 2-5 options like payment methods or shipping preferences.

Text Input

Standard single-line text fields with support for labels, placeholders, helper text, and validation states.

This will be displayed on your invoices
Please enter a valid tax ID format
License verified
This field is automatically assigned 
<!-- Default Input -->
<div class="input-group">
  <label class="input-label" for="company">Company Name</label>
  <input type="text" id="company" class="input-field input-md" placeholder="Enter company name">
  <span class="input-helper">This will be displayed on your invoices</span>
</div>

<!-- Required Input -->
<div class="input-group">
  <label class="input-label input-label-required" for="email">Email Address</label>
  <input type="email" id="email" class="input-field input-md" placeholder="email@example.com">
</div>

<!-- Error State -->
<div class="input-group">
  <label class="input-label" for="tax-id">Tax ID</label>
  <input type="text" id="tax-id" class="input-field input-md input-error" value="Invalid">
  <span class="input-error-message">Please enter a valid tax ID format</span>
</div>

<!-- Success State -->
<div class="input-group">
  <label class="input-label" for="license">Business License</label>
  <input type="text" id="license" class="input-field input-md input-success" value="BL-2024-1234">
  <span class="input-success-message">License verified</span>
</div>

<!-- Disabled -->
<div class="input-group">
  <label class="input-label" for="account-id">Account ID</label>
  <input type="text" id="account-id" class="input-field input-md" disabled>
  <span class="input-helper">This field is automatically assigned</span>
</div>
const TextInput = ({ label, id, type = 'text', required, error, success, helper, disabled, ...props }) => (
  <div className="input-group">
    <label className={`input-label ${required ? 'input-label-required' : ''}`} htmlFor={id}>
      {label}
    </label>
    <input
      type={type}
      id={id}
      className={`input-field input-md ${error ? 'input-error' : ''} ${success ? 'input-success' : ''}`}
      disabled={disabled}
      {...props}
    />
    {error && <span className="input-error-message">{error}</span>}
    {success && <span className="input-success-message">{success}</span>}
    {helper && <span className="input-helper">{helper}</span>}
  </div>
);

// Usage
<TextInput
  label="Company Name"
  id="company"
  placeholder="Enter company name"
  helper="This will be displayed on your invoices"
/>

      <option value="fresh">Fresh Seafood</option>

      <option value="processed">Processed Goods</option>
    </select>
    <span class="select-icon">
      <svg width="20" height="20">...</svg>
    </span>
  </div>
</div>

<!-- Select Sizes -->
<select class="select-field select-sm">...</select>
<select class="select-field">...</select>         <!-- Default medium -->
<select class="select-field select-lg">...</select>
const Select = ({ label, id, options, helper, error, ...props }) => (
  <div className="input-group">
    {label && <label className="input-label" htmlFor={id}>{label}</label>}
    <div className="select-wrapper">
      <select
        id={id}
        className={`select-field ${error ? 'input-error' : ''}`}
        {...props}
      >
        {options.map(opt => (
          <option key={opt.value} value={opt.value}>{opt.label}</option>
        ))}
      </select>
      <span className="select-icon">
        <ChevronDownIcon />
      </span>
    </div>
    {error && <span className="input-error-message">{error}</span>}
    {helper && <span className="input-helper">{helper}</span>}
  </div>
);

// Usage
<Select
  label="Product Category"
  id="category"
  options={[
    { value: '', label: 'Select a category' },
    { value: 'fresh', label: 'Fresh Seafood' },
    { value: 'frozen', label: 'Frozen Products' }
  ]}
/>

      <span class="radio-label">Overnight Delivery (next day) - $35.00</span>
    </label>
    <label class="radio-wrapper">
      <input type="radio" name="shipping" class="radio-input" disabled>

    </label>
  </div>

const RadioGroup = ({ label, name, options, value, onChange }) => (
  <div className="input-group">
    {label && <label className="input-label">{label}</label>}
    <div className="radio-group">
      {options.map(option => (
        <label key={option.value} className="radio-wrapper">
          <input
            type="radio"
            name={name}
            className="radio-input"
            value={option.value}
            checked={value === option.value}
            disabled={option.disabled}
            onChange={() => onChange(option.value)}
          />
          <span className="radio-label">{option.label}</span>
        </label>
      ))}
    </div>
  </div>
);

// Usage
<RadioGroup
  label="Shipping Method"
  name="shipping"
  value={shippingMethod}
  onChange={setShippingMethod}
  options={[
    { value: 'standard', label: 'Standard Delivery (3-5 days) - Free' },
    { value: 'express', label: 'Express Delivery (1-2 days) - $15.00' },
    { value: 'overnight', label: 'Overnight (next day) - $35.00' },
    { value: 'same-day', label: 'Same-Day (unavailable)', disabled: true }
  ]}
/>
