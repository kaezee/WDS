Modal

Overlay dialogs for confirmations, warnings, forms, and critical user decisions.

Usage & Purpose

Use modals when: Requiring user confirmation, displaying critical warnings, collecting focused input, or showing content that needs full attention without navigation.

Don't use modals for: Non-critical information (use snackbar), complex forms (use full page), or frequent interruptions that disrupt workflow.

Default Modal

Standard modal for general purpose dialogs, forms, and content display.

HTML + CSS 
<!-- Modal Overlay -->
<div class="modal-overlay hidden" id="my-modal">
  <div class="modal modal-md">
    <!-- Header -->
    <div class="modal-header">
      <h2 class="modal-title">Update Order Status</h2>
      <button class="modal-close" onclick="closeModal('my-modal')">
        <span class="modal-close-icon"></span>
      </button>
    </div>
    
    <!-- Body -->
    <div class="modal-body">
      <p>Are you sure you want to mark this order as shipped? This will notify the buyer and update the tracking information.</p>
    </div>
    
    <!-- Footer -->
    <div class="modal-footer">
      <button class="btn btn-md btn-secondary-outline" onclick="closeModal('my-modal')">Cancel</button>
      <button class="btn btn-md btn-primary">Confirm</button>
    </div>
  </div>
</div>

<!-- JavaScript -->
<script>
function openModal(id) {
  document.getElementById(id).classList.remove('hidden');
  document.body.classList.add('modal-open');
}

function closeModal(id) {
  document.getElementById(id).classList.add('hidden');
  document.body.classList.remove('modal-open');
}
</script>
JSX 
const Modal = ({ isOpen, onClose, title, children, onConfirm, size = 'md' }) => {
  if (!isOpen) return null;

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className={`modal modal-${size}`} onClick={(e) => e.stopPropagation()}>
        <div className="modal-header">
          <h2 className="modal-title">{title}</h2>
          <button className="modal-close" onClick={onClose}>
            <span className="modal-close-icon"></span>
          </button>
        </div>
        
        <div className="modal-body">
          {children}
        </div>
        
        <div className="modal-footer">
          <button className="btn btn-md btn-secondary-outline" onClick={onClose}>
            Cancel
          </button>
          <button className="btn btn-md btn-primary" onClick={onConfirm}>
            Confirm
          </button>
        </div>
      </div>
    </div>
  );
};

// Usage
<Modal
  isOpen={isModalOpen}
  onClose={() => setIsModalOpen(false)}
  onConfirm={handleConfirm}
  title="Update Order Status"
  size="md"
>
  <p>Are you sure you want to mark this order as shipped?</p>
</Modal>
